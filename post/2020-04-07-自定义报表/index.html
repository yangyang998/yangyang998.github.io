  • 2.为下拉框中添加数据

      with  ADOQuery1 do
   begin
      close;
      sql.clear;
      sql.add('SELECT [列] FROM [表]  ');  
      open;
      ComboBox1.items.clear;
      ComboBox1.items.add('');
      while not eof do
         begin
            ComboBox1.items.add(fieldbyname('[列] ').asstring);    
              next;                         
         end;                       
     end;

          end;
          boxt:=box;                            
     end;            
end;
